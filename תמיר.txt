            case ROTATE_CCW:
                velocities = velocity*0.8;
                speeds = -1.0;
                break;
        }

        drv.SetDirection(speeds > 0);
        
	if (Math.abs(speeds) > 0.0000001) {
                drv.move(Math.abs(speeds), velocities, distance, smooth, new Motor.MotorAdvCallback() { // הורדתי את המשתנה איי צריך להוריד גם מהפונקציה של מוטור זה המספר מנוע.
                    @Override
                    public boolean stop() {
                        return engineCallbacks.getEmergencyStop() || !condition.Condition();
                    }

                    @Override
                    public void OnComplete( long totalSteps, int totalDelay, double velocity) { // הורדתי את המשתנה איי צריך להוריד גם מהפונקציה של מוטור זה המספר מנוע.
                        OnAllComplete(totalSteps, totalDelay, velocity);
                    }
                });
            } else {
                complete = true;
            }
   
        if (wait) {
            while (getMovementState() == MovementState.MOVING) {
                Thread.sleep(10);
            }
